"Comparo los elementos de un vector para saber si están ordenados"
(dato
$?z
?x&: (numberp ?x)
?y&: (> ?x ?y)
$?w
=>
(printout t "El vector "$?z" "?x" "?y" "$?w" está ordenado"crlf)
)
